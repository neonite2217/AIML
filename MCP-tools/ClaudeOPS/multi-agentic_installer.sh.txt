# Detect system and Python
detect_system() {
    print_step "Detecting system configuration..."
    
    # Detect OS
    case "$(uname -s)" in
        Darwin) OS="macos" ;;
        Linux) OS="linux" ;;
        CYGWIN*|MINGW*|MSYS*) OS="windows" ;;
        *) print_error "Unsupported OS"; exit 1 ;;
    esac
    
    # Detect Python
    if command -v python3 >/dev/null 2>&1; then
        PYTHON_CMD="python3"
    elif command -v python >/dev/null 2>&1; then
        PYTHON_CMD="python"
    else
        print_error "Python not found. Please install Python 3.8+"
        exit 1
    fi
    
    # Check Python version
    python_version=$($PYTHON_CMD -c "import sys; print(f'{sys.version_info.major}.{sys.version_info.minor}')")
    if [[ $(echo "$python_version >= 3.8" | bc -l 2>/dev/null || echo "0") == "1" ]] || [[ "$python_version" > "3.7" ]]; then
        print_success "Python $python_version detected ($PYTHON_CMD)"
    else
        print_error "Python 3.8+ required, found $python_version"
        exit 1
    fi
    
    print_success "System: $OS, Python: $python_version"
}

@mcp.tool()
def system_info() -> str:
    """Get comprehensive system information"""
    try:
        cpu_percent = psutil.cpu_percent(interval=1)
        memory = psutil.virtual_memory()
        disk = psutil.disk_usage('/')
        
        info = {
            "system": {
                "platform": sys.platform,
                "python_version": sys.version,
                "cpu_percent": cpu_percent,
                "memory_percent": memory.percent,
                "disk_percent": disk.percent
            },
            "server": {
                "pid": os.getpid(),
                "working_dir": str(PROJECT_DIR),
                "uptime": str(datetime.now())
            }
        }
        
        return json.dumps(info, indent=2)
    except Exception as e:
        return f"Error getting system info: {e}"

@mcp.tool()
def execute_command(command: str, working_dir: Optional[str] = None) -> str:
    """Execute system command safely"""
    try:
        if working_dir:
            cwd = Path(working_dir)
            if not cwd.exists():
                return f"Error: Directory {working_dir} does not exist"
        else:
            cwd = PROJECT_DIR
        
        # Security check - basic command validation
        dangerous_commands = ['rm -rf /', 'format', 'del /f /s /q', 'shutdown', 'reboot']
        if any(dangerous in command.lower() for dangerous in dangerous_commands):
            return "Error: Dangerous command blocked"
        
        result = subprocess.run(
            command,
            shell=True,
            cwd=cwd,
            capture_output=True,
            text=True,
            timeout=30
        )
        
        output = f"Exit code: {result.returncode}\n"
        if result.stdout:
            output += f"STDOUT:\n{result.stdout}\n"
        if result.stderr:
            output += f"STDERR:\n{result.stderr}\n"
        
        return output
    except subprocess.TimeoutExpired:
        return "Error: Command timed out (30s limit)"
    except Exception as e:
        return f"Error executing command: {e}"

@mcp.tool()
def list_directory(path: str = ".") -> str:
    """List directory contents with details"""
    try:
        target_path = Path(path)
        if not target_path.exists():
            return f"Error: Path {path} does not exist"
        
        if not target_path.is_dir():
            return f"Error: {path} is not a directory"
        
        items = []
        for item in sorted(target_path.iterdir()):
            try:
                stat = item.stat()
                size = stat.st_size if item.is_file() else "DIR"
                modified = datetime.fromtimestamp(stat.st_mtime).strftime("%Y-%m-%d %H:%M")
                items.append(f"{item.name:<30} {str(size):<10} {modified}")
            except:
                items.append(f"{item.name:<30} {'ERROR':<10} {'N/A'}")
        
        return f"Directory: {path}\n" + "\n".join(items) if items else f"Directory {path} is empty"
    except Exception as e:
        return f"Error listing directory: {e}"

@mcp.tool()
def read_file(filepath: str, max_lines: int = 100) -> str:
    """Read file contents with line limit"""
    try:
        file_path = Path(filepath)
        if not file_path.exists():
            return f"Error: File {filepath} does not exist"
        
        if not file_path.is_file():
            return f"Error: {filepath} is not a file"
        
        with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
            lines = f.readlines()
        
        if len(lines) > max_lines:
            content = ''.join(lines[:max_lines])
            content += f"\n... (truncated, showing first {max_lines} lines of {len(lines)})"
        else:
            content = ''.join(lines)
        
        return f"File: {filepath}\nLines: {len(lines)}\n\n{content}"
    except Exception as e:
        return f"Error reading file: {e}"

@mcp.tool()
def write_file(filepath: str, content: str, append: bool = False) -> str:
    """Write content to file"""
    try:
        file_path = Path(filepath)
        
        # Security check - only allow writing in project directory or home
        if not (str(file_path).startswith(str(PROJECT_DIR)) or str(file_path).startswith(str(HOME_DIR))):
            return "Error: Can only write files in project or home directory"
        
        mode = 'a' if append else 'w'
        with open(file_path, mode, encoding='utf-8') as f:
            f.write(content)
        
        action = "appended to" if append else "written to"
        return f"Content {action} {filepath} ({len(content)} characters)"
    except Exception as e:
        return f"Error writing file: {e}"

if __name__ == "__main__":
    logger.info("Starting LocalOps MCP Server...")
    mcp.run()
EOF

    print_success "Enhanced MCP server created"

# Auto-configure clients
configure_clients() {
    print_header "Client Configuration"
    echo "Select clients to configure automatically:"
    echo
    
    local clients_to_configure=()
    
    # Check which clients are available and ask user
    for client in "${!CLIENT_CONFIGS[@]}"; do
        echo -n "Configure ${CLIENT_CONFIGS[$client]}? (y/N): "
        read -r response
        if [[ $response =~ ^[Yy]$ ]]; then
            clients_to_configure+=("$client")
        fi
    done
    
    # Configure selected clients
    for client in "${clients_to_configure[@]}"; do
        configure_client "$client"
    done
}

# Create AI desktop client
create_ai_client() {
    print_step "Creating minimal AI desktop client..."
    
    mkdir -p clients/desktop
    
    # Python-based desktop client
    cat > clients/desktop/ai_client.py << 'EOF'
#!/usr/bin/env python3
"""
Minimal AI Desktop Client
Supports LocalAI (Ollama), APIs, and MCP servers
"""
import asyncio
import json
import sys
import os
from pathlib import Path
from typing import Dict, List, Optional
import subprocess
import httpx
from rich.console import Console
from rich.prompt import Prompt
from rich.panel import Panel
from rich.markdown import Markdown

console = Console()

class AIClient:
    def __init__(self):
        self.config = self.load_config()
        self.mcp_servers = {}
        self.current_model = None
        
    def load_config(self) -> Dict:
        """Load client configuration"""
        config_file = Path(__file__).parent / "config.json"
        if config_file.exists():
            with open(config_file) as f:
                return json.load(f)
        
        # Default configuration
        default_config = {
            "models": {
                "ollama": {
                    "enabled": True,
                    "url": "http://localhost:11434",
                    "models": ["llama2", "codellama", "mistral"]
                },
                "openai": {
                    "enabled": False,
                    "api_key": "",
                    "models": ["gpt-3.5-turbo", "gpt-4"]
                }
            },
            "mcp_servers": {
                "localops": {
                    "command": ["python", "../../localops_server.py"],
                    "enabled": True
                }
            }
        }
        
        with open(config_file, 'w') as f:
            json.dump(default_config, f, indent=2)
        
        return default_config
    
    async def start_mcp_servers(self):
        """Start configured MCP servers"""
        for name, config in self.config["mcp_servers"].items():
            if config.get("enabled", False):
                try:
                    process = await asyncio.create_subprocess_exec(
                        *config["command"],
                        stdin=asyncio.subprocess.PIPE,
                        stdout=asyncio.subprocess.PIPE,
                        stderr=asyncio.subprocess.PIPE
                    )
                    self.mcp_servers[name] = process
                    console.print(f"✅ Started MCP server: {name}")
                except Exception as e:
                    console.print(f"❌ Failed to start {name}: {e}")
    
    async def query_ollama(self, prompt: str, model: str = "llama2") -> str:
        """Query Ollama API"""
        try:
            async with httpx.AsyncClient() as client:
                response = await client.post(
                    f"{self.config['models']['ollama']['url']}/api/generate",
                    json={
                        "model": model,
                        "prompt": prompt,
                        "stream": False
                    },
                    timeout=30.0
                )
                result = response.json()
                return result.get("response", "No response")
        except Exception as e:
            return f"Error querying Ollama: {e}"
    
    async def run_interactive(self):
        """Run interactive chat session"""
        console.print(Panel.fit("🤖 AI Desktop Client", style="bold blue"))
        console.print("Type 'help' for commands, 'quit' to exit")
        
        await self.start_mcp_servers()
        
        while True:
            try:
                user_input = Prompt.ask("\n[bold green]You[/bold green]")
                
                if user_input.lower() in ['quit', 'exit']:
                    break
                elif user_input.lower() == 'help':
                    self.show_help()
                elif user_input.startswith('/'):
                    await self.handle_command(user_input[1:])
                else:
                    # Send to AI model
                    console.print("\n[bold blue]AI[/bold blue]:")
                    response = await self.query_ollama(user_input)
                    console.print(Markdown(response))
                    
            except KeyboardInterrupt:
                break
            except Exception as e:
                console.print(f"[red]Error: {e}[/red]")
        
        await self.cleanup()
    
    def show_help(self):
        """Show help information"""
        help_text = """
**Available Commands:**
- `/models` - List available models
- `/switch <model>` - Switch AI model
- `/mcp <server> <tool> <args>` - Call MCP tool
- `/config` - Show configuration
- `help` - Show this help
- `quit` - Exit client
        """
        console.print(Markdown(help_text))
    
    async def handle_command(self, command: str):
        """Handle slash commands"""
        parts = command.split()
        cmd = parts[0]
        
        if cmd == "models":
            console.print("Available models:")
            for provider, config in self.config["models"].items():
                if config.get("enabled"):
                    for model in config.get("models", []):
                        console.print(f"  - {provider}:{model}")
        
        elif cmd == "config":
            console.print(json.dumps(self.config, indent=2))
        
        else:
            console.print(f"Unknown command: {cmd}")
    
    async def cleanup(self):
        """Cleanup resources"""
        for name, process in self.mcp_servers.items():
            try:
                process.terminate()
                await process.wait()
                console.print(f"✅ Stopped MCP server: {name}")
            except:
                pass

if __name__ == "__main__":
    client = AIClient()
    asyncio.run(client.run_interactive())
EOF

    print_success "AI desktop client created"
}# Cr

            if err := cmd.Start(); err != nil {

func (c *AIClient) queryOllama(prompt, model string) (string, error) {
    payload := map[string]interface{}{
        "model":  model,
        "prompt": prompt,
        "stream": false,
    }
    
    jsonData, _ := json.Marshal(payload)
    
    client := &http.Client{Timeout: 30 * time.Second}
    resp, err := client.Post(
        c.config.Models.Ollama.URL+"/api/generate",
        "application/json",
        strings.NewReader(string(jsonData)),
    )
    if err != nil {
        return "", err
    }
    defer resp.Body.Close()
    
    var result map[string]interface{}
    if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
        return "", err
    }
    
    if response, ok := result["response"].(string); ok {
        return response, nil
    }
    
    return "No response", nil
}

use serde::{Deserialize, Serialize};
use std::collections::HashMap;
use std::io::{self, Write};
use std::process::{Child, Command};
use tokio::time::{timeout, Duration};

#[derive(Debug, Serialize, Deserialize)]
struct Config {
    models: Models,
    mcp_servers: HashMap<String, MCPServerConfig>,
}

#[derive(Debug, Serialize, Deserialize)]
struct Models {
    ollama: OllamaConfig,
}

#[derive(Debug, Serialize, Deserialize)]
struct OllamaConfig {
    enabled: bool,
    url: String,
    models: Vec<String>,
}

#[derive(Debug, Serialize, Deserialize)]
struct MCPServerConfig {
    command: Vec<String>,
    enabled: bool,
}

#[derive(Debug, Serialize)]
struct OllamaRequest {
    model: String,
    prompt: String,
    stream: bool,
}

#[derive(Debug, Deserialize)]
struct OllamaResponse {
    response: String,
}

struct AIClient {
    config: Config,
    mcp_servers: HashMap<String, Child>,
}

impl AIClient {
    fn new() -> Self {
        let config = Config {
            models: Models {
                ollama: OllamaConfig {
                    enabled: true,
                    url: "http://localhost:11434".to_string(),
                    models: vec!["llama2".to_string(), "codellama".to_string()],
                },
            },
            mcp_servers: {
                let mut servers = HashMap::new();
                servers.insert(
                    "localops".to_string(),
                    MCPServerConfig {
                        command: vec!["python".to_string(), "../../localops_server.py".to_string()],
                        enabled: true,
                    },
                );
                servers
            },
        };

        Self {
            config,
            mcp_servers: HashMap::new(),
        }
    }

    fn start_mcp_servers(&mut self) {
        for (name, config) in &self.config.mcp_servers {
            if config.enabled {
                match Command::new(&config.command[0])
                    .args(&config.command[1..])
                    .spawn()
                {
                    Ok(child) => {
                        println!("✅ Started MCP server: {}", name);
                        self.mcp_servers.insert(name.clone(), child);
                    }
                    Err(e) => {
                        println!("❌ Failed to start MCP server {}: {}", name, e);
                    }
                }
            }
        }
    }

    async fn query_ollama(&self, prompt: &str, model: &str) -> Result<String, Box<dyn std::error::Error>> {
        let client = reqwest::Client::new();
        let request = OllamaRequest {
            model: model.to_string(),
            prompt: prompt.to_string(),
            stream: false,
        };

        let response = timeout(
            Duration::from_secs(30),
            client
                .post(&format!("{}/api/generate", self.config.models.ollama.url))
                .json(&request)
                .send(),
        )
        .await??;

        let result: OllamaResponse = response.json().await?;
        Ok(result.response)
    }

    async fn run_interactive(&mut self) {
        println!("🤖 AI Desktop Client (Rust)");
        println!("Type 'help' for commands, 'quit' to exit");

        self.start_mcp_servers();

        loop {
            print!("\nYou: ");
            io::stdout().flush().unwrap();

            let mut input = String::new();
            if io::stdin().read_line(&mut input).is_err() {
                break;
            }

            let input = input.trim();

            match input {
                "quit" | "exit" => break,
                "help" => self.show_help(),
                cmd if cmd.starts_with('/') => self.handle_command(&cmd[1..]),
                _ => {
                    print!("AI: ");
                    match self.query_ollama(input, "llama2").await {
                        Ok(response) => println!("{}", response),
                        Err(e) => println!("Error: {}", e),
                    }
                }
            }
        }

        self.cleanup();
    }

    fn show_help(&self) {
        println!(
            r#"
Available Commands:
- /models - List available models
- help - Show this help
- quit - Exit client
"#
        );
    }

    fn handle_command(&self, command: &str) {
        let parts: Vec<&str> = command.split_whitespace().collect();
        if parts.is_empty() {
            return;
        }

        match parts[0] {
            "models" => {
                println!("Available models:");
                for model in &self.config.models.ollama.models {
                    println!("  - ollama:{}", model);
                }
            }
            _ => println!("Unknown command: {}", parts[0]),
        }
    }

    fn cleanup(&mut self) {
        for (name, child) in &mut self.mcp_servers {
            if let Err(e) = child.kill() {
                println!("Failed to stop MCP server {}: {}", name, e);
            } else {
                println!("✅ Stopped MCP server: {}", name);
            }
        }
    }
}

async fn main() {

    client.run_interactive().await;
