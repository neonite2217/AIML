# Print functions
print_step() { echo -e "${BLUE}→ $1${NC}"; }
print_success() { echo -e "${GREEN}✅ $1${NC}"; }
print_warning() { echo -e "${YELLOW}⚠️  $1${NC}"; }
print_error() { echo -e "${RED}❌ $1${NC}"; }
print_info() { echo -e "${BLUE}ℹ️  $1${NC}"; }

# Startup graffiti
show_startup_banner() {
    clear
    echo -e "${BOLD}${BLUE}"

    echo "║             Claude Ops MCP Server                   ║"

# Loading animation
show_loading() {
    local message="$1"
    local duration="${2:-3}"
    
    echo -n "$message"
    for i in $(seq 1 $duration); do
        echo -n "."
        sleep 1
    done
    echo " ✅"
}

# Enhanced prerequisite checking
check_prerequisites() {
    print_step "Checking prerequisites..."
    
    # Detect Python command
    if command -v python3 >/dev/null 2>&1; then
        PYTHON_CMD="python3"
    elif command -v python >/dev/null 2>&1; then
        PYTHON_CMD="python"
    else
        print_error "Python not found. Please install Python 3.8+"
        exit 1
    fi
    
    # Check Python version
    python_version_full=$($PYTHON_CMD --version 2>&1)
    print_info "Detected Python: $python_version_full"
    
    version_check_result=$($PYTHON_CMD -c "
import sys
major, minor = sys.version_info[:2]
required_major, required_minor = 3, 8
if (major, minor) >= (required_major, required_minor):
    print('OK')
    print(f'{major}.{minor}')
else:
    print('FAIL')
    print(f'{major}.{minor}')
")
    
    version_status=$(echo "$version_check_result" | head -n1)
    detected_version=$(echo "$version_check_result" | tail -n1)
    
    if [[ "$version_status" == "OK" ]]; then
        print_success "Python $detected_version found ($PYTHON_CMD) - meets requirement (3.8+)"
    else
        print_error "Python $detected_version found, but 3.8+ required"
        exit 1
    fi
    
    # Check Docker
    if ! command -v docker >/dev/null 2>&1; then
        print_error "Docker not found. Please install Docker Desktop"
        exit 1
    fi
    
    if ! docker info >/dev/null 2>&1; then
        print_warning "Docker found but not running"
        case "$(uname -s)" in
            Darwin) open -a Docker 2>/dev/null || true ;;
            Linux) sudo systemctl start docker 2>/dev/null || true ;;
        esac
        
        print_info "Waiting for Docker to start..."
        for i in {1..30}; do
            if docker info >/dev/null 2>&1; then
                break
            fi
            sleep 1
            echo -n "."
        done
        echo
        
        if ! docker info >/dev/null 2>&1; then
            print_error "Failed to start Docker. Please start it manually."
            exit 1
        fi
    fi
    
    print_success "Docker is running"
    
    # Check pip
    if ! $PYTHON_CMD -m pip --version >/dev/null 2>&1; then
        print_error "pip not found. Please install pip"
        exit 1
    fi
    
    print_success "Prerequisites check passed"
}

# Create project structure in current directory
create_project() {
    print_step "Creating claudeops project in current directory..."
    
    if [[ -d "$PROJECT_DIR" ]]; then
        print_warning "Directory $PROJECT_DIR already exists"
        read -p "Remove existing directory? (y/N): " -r
        if [[ $REPLY =~ ^[Yy]$ ]]; then
            rm -rf "$PROJECT_DIR"
            print_info "Removed existing directory"
        else
            print_error "Cannot proceed with existing directory"
            exit 1
        fi
    fi
    
    mkdir -p "$PROJECT_DIR"
    cd "$PROJECT_DIR"
    
    print_success "Project directory created: $PROJECT_DIR"
}

# Create virtual environment and install requirements in the same place
setup_python_environment() {
    print_step "Setting up Python environment in project directory..."
    
    # Create requirements.txt first
    cat > requirements.txt << 'EOF'
mcp>=1.2.0
httpx>=0.25.0
psutil>=5.9.0
fastmcp>=0.2.0
EOF
    print_success "requirements.txt created"
    
    # Remove any existing broken venv
    if [[ -d "venv" ]]; then
        print_info "Removing existing virtual environment..."
        rm -rf venv
    fi
    
    # Create virtual environment in project directory
    print_step "Creating virtual environment"
    show_loading "Setting up Python virtual environment" 2
    if $PYTHON_CMD -m venv venv; then
        print_success "Virtual environment created in $PROJECT_DIR/venv"
    else
        print_error "Failed to create virtual environment"
        exit 1
    fi
    
    # Determine venv Python path
    if [[ -f "venv/bin/python" ]]; then
        VENV_PYTHON="$PROJECT_DIR/venv/bin/python"
        VENV_ACTIVATE="$PROJECT_DIR/venv/bin/activate"
        print_success "Virtual environment setup (Unix/macOS)"
    elif [[ -f "venv/Scripts/python.exe" ]]; then
        VENV_PYTHON="$PROJECT_DIR/venv/Scripts/python.exe"
        VENV_ACTIVATE="$PROJECT_DIR/venv/Scripts/activate"
        print_success "Virtual environment setup (Windows)"
    else
        print_error "Virtual environment created but Python executable not found"
        exit 1
    fi
    
    print_info "Using venv Python: $VENV_PYTHON"
    
    # Verify venv Python version
    venv_version=$($VENV_PYTHON --version 2>&1)
    print_info "Virtual environment Python: $venv_version"
    
    # Upgrade pip in venv
    print_step "Upgrading pip in virtual environment..."
    $VENV_PYTHON -m pip install --upgrade pip
    
    # Install requirements in venv
    print_step "Installing packages in virtual environment"
    show_loading "Installing MCP dependencies" 3
    if $VENV_PYTHON -m pip install -r requirements.txt >/dev/null 2>&1; then
        print_success "All packages installed successfully in venv"
    else
        print_error "Failed to install packages"
        print_info "Trying individual package installation..."
        
        packages=("mcp>=1.2.0" "httpx>=0.25.0" "psutil>=5.9.0" "fastmcp>=0.2.0")
        failed_packages=()
        
        for package in "${packages[@]}"; do
            print_info "Installing $package..."
            if $VENV_PYTHON -m pip install "$package"; then
                print_success "Installed: $package"
            else
                print_warning "Failed to install: $package"
                failed_packages+=("$package")
            fi
        done
        
        if [ ${#failed_packages[@]} -gt 0 ]; then
            print_error "Failed packages: ${failed_packages[*]}"
        fi
    fi
    
    # Validate installations in venv
    print_step "Validating package installations in venv..."
    if $VENV_PYTHON -c "
import sys
required_packages = ['mcp', 'httpx', 'psutil', 'fastmcp']
missing = []
installed = []

print('Checking packages in virtual environment...')
print(f'Python executable: {sys.executable}')
print(f'Python path: {sys.path[0]}')
print()

for package in required_packages:
    try:
        __import__(package)
        installed.append(package)
        print(f'✓ {package}')
    except ImportError as e:
        missing.append(package)
        print(f'✗ {package} - MISSING ({e})')

print(f'\\nInstalled: {len(installed)}/{len(required_packages)}')
if missing:
    print(f'Missing packages: {missing}')
    sys.exit(1)
else:
    print('All required packages are available in venv!')
    sys.exit(0)
"; then
        print_success "Package validation passed in virtual environment"
    else
        print_error "Package validation failed in virtual environment"
        exit 1
    fi
}

# Ensure directories exist
CONFIG_DIR.mkdir(parents=True, exist_ok=True)
WORKSPACE_DIR.mkdir(parents=True, exist_ok=True)

# Server state management
def save_pid():
    """Save server PID for management"""
    with open(PID_FILE, 'w') as f:
        f.write(str(os.getpid()))
    logger.info(f"Server PID {os.getpid()} saved to {PID_FILE}")

def cleanup_pid():
    """Remove PID file on shutdown"""
    if PID_FILE.exists():
        PID_FILE.unlink()
        logger.info("PID file cleaned up")

def signal_handler(signum, frame):
    """Handle shutdown signals"""
    logger.info(f"Received signal {signum}, shutting down...")
    cleanup_pid()
    sys.exit(0)

# Register cleanup handlers
atexit.register(cleanup_pid)
signal.signal(signal.SIGTERM, signal_handler)
signal.signal(signal.SIGINT, signal_handler)

def is_safe_path(path_str):
    """Check if path is within allowed directories - Full home directory access"""
    try:
        path = Path(path_str).resolve()
        # Allow full access to home directory and subdirectories
        home_str = str(USER_HOME)
        path_str = str(path)
        
        # Allow access to entire home directory
        if path_str.startswith(home_str):
            return True
            
        # Also allow common system paths for reading (but not writing)
        safe_read_paths = ['/usr', '/opt', '/etc', '/var/log', '/tmp']
        for safe_path in safe_read_paths:
            if path_str.startswith(safe_path):
                return True
                
        return False
    except:
        return False

def can_write_path(path_str):
    """Check if path allows write operations - More restrictive than read"""
    try:
        path = Path(path_str).resolve()
        home_str = str(USER_HOME)
        path_str = str(path)
        
        # Only allow writing within home directory
        return path_str.startswith(home_str)
    except:
        return False

def is_command_allowed(command):
    """Check if command is allowed"""
    command = command.strip()
    for category, patterns in ALLOWED_COMMANDS.items():
        for pattern in patterns:
            if command.startswith(pattern):
                return True
    return False

@mcp.tool()
def read_file(filepath: str, encoding: str = "auto") -> str:
    """Read any file type with automatic encoding detection"""
    try:
        if not is_safe_path(filepath):
            return "❌ Error: Access denied - path outside allowed directories"
        
        path = Path(filepath)
        if not path.exists():
            return f"❌ Error: File not found: {filepath}"
        
        if not path.is_file():
            return f"❌ Error: Path is not a file: {filepath}"
        
        # Get file info
        stat_info = path.stat()
        file_size = stat_info.st_size
        file_ext = path.suffix.lower()
        
        # Handle different file types
        if file_ext in ['.jpg', '.jpeg', '.png', '.gif', '.bmp', '.ico', '.svg']:
            return f"📷 Image File: {filepath}\n📊 Size: {file_size} bytes\n⚠️ Binary image file - cannot display text content"
        
        if file_ext in ['.pdf', '.doc', '.docx', '.xls', '.xlsx', '.ppt', '.pptx']:
            return f"📄 Document File: {filepath}\n📊 Size: {file_size} bytes\n⚠️ Binary document file - use specialized tools to read content"
        
        if file_ext in ['.zip', '.tar', '.gz', '.bz2', '.7z', '.rar']:
            return f"📦 Archive File: {filepath}\n📊 Size: {file_size} bytes\n⚠️ Compressed archive - use extract_archive tool to access contents"
        
        if file_ext in ['.exe', '.dll', '.so', '.dylib', '.bin']:
            return f"⚙️ Binary File: {filepath}\n📊 Size: {file_size} bytes\n⚠️ Executable binary file - cannot display as text"
        
        # Try to read as text with encoding detection
        encodings_to_try = ['utf-8', 'utf-16', 'latin-1', 'cp1252', 'ascii'] if encoding == "auto" else [encoding]
        
        content = None
        used_encoding = None
        
        for enc in encodings_to_try:
            try:
                with open(path, 'r', encoding=enc) as f:
                    content = f.read()
                    used_encoding = enc
                    break
            except UnicodeDecodeError:
                continue
        
        if content is None:
            # Try binary read for hex dump of small files
            if file_size < 1024:
                with open(path, 'rb') as f:
                    binary_content = f.read()
                    hex_content = ' '.join(f'{b:02x}' for b in binary_content)
                    return f"🔧 Binary File: {filepath}\n📊 Size: {file_size} bytes\n🔍 Hex dump (first 1KB):\n{hex_content}"
            else:
                return f"❌ Error: Cannot decode file as text with any common encoding: {filepath}\n📊 Size: {file_size} bytes"
        
        # Truncate very large files
        if len(content) > 50000:
            content = content[:50000] + f"\n\n... [File truncated - showing first 50,000 characters of {len(content)} total]"
        
        file_type = "📄 Text File"
        if file_ext in ['.py']:
            file_type = "🐍 Python File"
        elif file_ext in ['.js', '.ts', '.jsx', '.tsx']:
            file_type = "📜 JavaScript/TypeScript File"
        elif file_ext in ['.java']:
            file_type = "☕ Java File"
        elif file_ext in ['.sh', '.bash', '.zsh']:
            file_type = "🔧 Shell Script"
        elif file_ext in ['.html', '.htm']:
            file_type = "🌐 HTML File"
        elif file_ext in ['.css']:
            file_type = "🎨 CSS File"
        elif file_ext in ['.json']:
            file_type = "📋 JSON File"
        elif file_ext in ['.xml']:
            file_type = "📋 XML File"
        elif file_ext in ['.md', '.markdown']:
            file_type = "📝 Markdown File"
        elif file_ext in ['.yml', '.yaml']:
            file_type = "⚙️ YAML File"
        elif file_ext in ['.ini', '.conf', '.config']:
            file_type = "⚙️ Configuration File"
        elif file_ext in ['.log']:
            file_type = "📊 Log File"
        
        return f"{file_type}: {filepath}\n📊 Size: {file_size} bytes\n🔤 Encoding: {used_encoding}\n\n{content}"
        
    except Exception as e:
        return f"❌ Error: {str(e)}"

@mcp.tool()

    try:

        file_ext = path.suffix.lower()
        
        # Handle different write modes
        write_mode = mode
        if mode not in ['w', 'a', 'x']:
            write_mode = 'w'
        
        # Write the file
        with open(path, write_mode, encoding=encoding) as f:
            f.write(content)
        
        # Get file info after writing
        stat_info = path.stat()
        file_size = stat_info.st_size
        
        # Determine file type for response
        file_type = "📄 Text File"
        if file_ext in ['.py']:
            file_type = "🐍 Python File"
        elif file_ext in ['.js', '.ts', '.jsx', '.tsx']:
            file_type = "📜 JavaScript/TypeScript File"
        elif file_ext in ['.java']:
            file_type = "☕ Java File"
        elif file_ext in ['.sh', '.bash', '.zsh']:
            file_type = "🔧 Shell Script"
            # Make shell scripts executable
            path.chmod(0o755)
        elif file_ext in ['.html', '.htm']:
            file_type = "🌐 HTML File"
        elif file_ext in ['.css']:
            file_type = "🎨 CSS File"
        elif file_ext in ['.json']:
            file_type = "📋 JSON File"
        elif file_ext in ['.xml']:
            file_type = "📋 XML File"
        elif file_ext in ['.md', '.markdown']:
            file_type = "📝 Markdown File"
        elif file_ext in ['.yml', '.yaml']:
            file_type = "⚙️ YAML File"
        elif file_ext in ['.ini', '.conf', '.config']:
            file_type = "⚙️ Configuration File"
        
        mode_desc = {"w": "Created/Overwritten", "a": "Appended to", "x": "Created (new)"}

    except FileExistsError:

@mcp.tool()

    """Execute allowed system command"""
    try:
        if not command.strip():
            return "Error: No command provided"
        
        if not is_command_allowed(command):
            return "Error: Command not allowed"
        
        result = subprocess.run(command, shell=True, capture_output=True, text=True, timeout=30)
        
        output = [f"Command: {command}", f"Exit code: {result.returncode}"]
        if result.stdout:
            output.append(f"Output:\n{result.stdout}")
        if result.stderr:
            output.append(f"Error:\n{result.stderr}")
        
        return "\n\n".join(output)
    except subprocess.TimeoutExpired:
        return f"Error: Command timed out: {command}"
    except Exception as e:
        return f"Error: {e}"

@mcp.tool()
def change_permissions(filepath: str, permissions: str) -> str:
    """Change file or directory permissions using chmod"""
    try:
        if not is_safe_path(filepath):
            return "❌ Error: Access denied - path outside allowed directories"
        
        path = Path(filepath)
        if not path.exists():
            return f"❌ Error: Path does not exist: {filepath}"
        
        # Validate permissions format (octal or symbolic)
        import re
        if not (re.match(r'^[0-7]{3,4}$', permissions) or re.match(r'^[ugoa]*[+-=][rwxXst]+$', permissions)):
            return f"❌ Error: Invalid permissions format: {permissions}. Use octal (755) or symbolic (u+x)"
        
        command = f"chmod {permissions} {filepath}"
        result = subprocess.run(command, shell=True, capture_output=True, text=True, timeout=30)
        
        if result.returncode == 0:
            # Get new permissions
            stat_result = subprocess.run(f"ls -la {filepath}", shell=True, capture_output=True, text=True)
            return f"✅ Permissions changed successfully\n📤 New permissions:\n{stat_result.stdout}"
        else:
            return f"❌ Failed to change permissions: {result.stderr}"
        
    except Exception as e:
        return f"❌ Error: {str(e)}"

        return f"❌ Error: {str(e)}"

@mcp.tool()
def extract_archive(archive_path: str, destination: str = ".") -> str:
    """Extract compressed archives"""
    try:
        if not is_safe_path(archive_path) or not is_safe_path(destination):
            return "❌ Error: Access denied - path outside allowed directories"
        
        archive = Path(archive_path)
        if not archive.exists():
            return f"❌ Error: Archive does not exist: {archive_path}"
        
        # Determine extraction command based on file extension
        if archive_path.endswith('.tar.gz') or archive_path.endswith('.tgz'):
            command = f"tar -xzf {archive_path} -C {destination}"
        elif archive_path.endswith('.tar.bz2'):
            command = f"tar -xjf {archive_path} -C {destination}"
        elif archive_path.endswith('.tar'):
            command = f"tar -xf {archive_path} -C {destination}"
        elif archive_path.endswith('.zip'):
            command = f"unzip {archive_path} -d {destination}"
        else:
            return f"❌ Error: Unsupported archive format: {archive_path}"
        
        result = subprocess.run(command, shell=True, capture_output=True, text=True, timeout=300)
        
        if result.returncode == 0:
            return f"✅ Extraction successful\n📁 Extracted to: {destination}\n📤 Output:\n{result.stdout}"
        else:
            return f"❌ Extraction failed: {result.stderr}"
        
    except Exception as e:
        return f"❌ Error: {str(e)}"

@mcp.tool()
def network_request(url: str, method: str = "GET", headers: str = "", data: str = "") -> str:
    """Make HTTP requests using curl"""
    try:
        # Build curl command
        command_parts = ["curl", "-s", "-w", "'\\nHTTP Status: %{http_code}\\nTotal Time: %{time_total}s\\n'"]
        
        if method.upper() != "GET":
            command_parts.extend(["-X", method.upper()])
        
        if headers:
            for header in headers.split(";"):
                if header.strip():
                    command_parts.extend(["-H", f"'{header.strip()}'"])
        
        if data and method.upper() in ["POST", "PUT", "PATCH"]:
            command_parts.extend(["-d", f"'{data}'"])
        
        command_parts.append(f"'{url}'")
        command = " ".join(command_parts)
        
        result = subprocess.run(command, shell=True, capture_output=True, text=True, timeout=60)
        
        output = [f"🌐 HTTP Request: {method.upper()} {url}"]
        if result.stdout:
            output.append(f"📤 Response:\n{result.stdout}")
        if result.stderr:
            output.append(f"⚠️ Error:\n{result.stderr}")
        
        return "\n\n".join(output)
        
    except Exception as e:
        return f"❌ Error: {str(e)}"

            "*.class", "*.o", "*.obj", "*.exe.stackdump", "core.*"
        ]
        
        temp_directories = [
            ".cache", "__pycache__", ".pytest_cache", "node_modules/.cache",
            ".npm", ".yarn/cache", ".composer/cache", ".gradle/caches",
            ".m2/repository", "Library/Caches", "Library/Logs",
            ".local/share/Trash", ".Trash", "Downloads/*.dmg", "Downloads/*.pkg"
        ]
        
        large_file_extensions = [
            ".dmg", ".pkg", ".iso", ".img", ".ova", ".vmdk", ".vdi",
            ".zip", ".tar.gz", ".tar.bz2", ".7z", ".rar"
        ] if aggressive else []
        
        duplicate_patterns = [
            "*copy*", "*Copy*", "* (1).*", "* (2).*", "*duplicate*"
        ] if aggressive else []
        
        # Start scanning from home directory
        scan_paths = [USER_HOME]
        
        cleanup_results.append("🧹 System Cleanup Analysis")
        cleanup_results.append("=" * 50)
        
        for base_path in scan_paths:
            cleanup_results.append(f"\n📁 Scanning: {base_path}")
            
            # Scan for temporary files
            cleanup_results.append("\n🗑️  Temporary Files:")
            for pattern in temp_patterns:
                try:
                    for file_path in Path(base_path).rglob(pattern):
                        if file_path.is_file():
                            size = file_path.stat().st_size
                            age_days = (datetime.now() - datetime.fromtimestamp(file_path.stat().st_mtime)).days
                            
                            if age_days > 7 or pattern in ["*.pyc", "*.pyo", "*.class"]:  # Old or compiled files
                                files_found += 1
                                total_size_saved += size
                                size_mb = size / (1024 * 1024)
                                
                                if not scan_only:
                                    file_path.unlink()
                                    status = "🗑️ DELETED"
                                else:
                                    status = "📋 FOUND"
                                
                                cleanup_results.append(f"  {status}: {file_path} ({size_mb:.2f} MB, {age_days} days old)")
                                
                                if files_found > 100:  # Limit output
                                    cleanup_results.append(f"  ... and more (limiting output)")
                                    break
                except PermissionError:
                    continue
                except Exception:
                    continue
            
            # Scan for temporary directories
            cleanup_results.append("\n📂 Temporary Directories:")
            for dir_pattern in temp_directories:
                try:
                    for dir_path in Path(base_path).rglob(dir_pattern):
                        if dir_path.is_dir():
                            try:
                                dir_size = sum(f.stat().st_size for f in dir_path.rglob('*') if f.is_file())
                                size_mb = dir_size / (1024 * 1024)
                                
                                if size_mb > 10:  # Only show directories > 10MB
                                    files_found += 1
                                    total_size_saved += dir_size
                                    
                                    if not scan_only and dir_pattern in [".cache", "__pycache__", ".pytest_cache"]:
                                        import shutil
                                        shutil.rmtree(dir_path)
                                        status = "🗑️ DELETED"
                                    else:
                                        status = "📋 FOUND"
                                    
                                    cleanup_results.append(f"  {status}: {dir_path} ({size_mb:.2f} MB)")
                            except:
                                continue
                except:
                    continue
            
            # Scan for large files (if aggressive)
            if aggressive:
                cleanup_results.append("\n📦 Large Files (>100MB):")
                try:
                    for file_path in Path(base_path).rglob('*'):
                        if file_path.is_file():
                            size = file_path.stat().st_size
                            size_mb = size / (1024 * 1024)
                            
                            if size_mb > 100:
                                age_days = (datetime.now() - datetime.fromtimestamp(file_path.stat().st_mtime)).days
                                cleanup_results.append(f"  📋 LARGE: {file_path} ({size_mb:.2f} MB, {age_days} days old)")
                                
                                if len(cleanup_results) > 200:  # Limit output
                                    break

            # Scan for potential duplicates (if aggressive)
            if aggressive:

                for pattern in duplicate_patterns:

        cleanup_results.append("\n🔒 Safety Notes:")
        cleanup_results.append("• Only scans/cleans within home directory")
        cleanup_results.append("• Preserves files newer than 7 days (except compiled files)")
        cleanup_results.append("• Aggressive mode shows large files and duplicates")
        cleanup_results.append("• Always backup important data before cleanup")
        
        return "\n".join(cleanup_results)
        
    except Exception as e:
        return f"❌ Error during cleanup: {str(e)}"

@mcp.tool()
def analyze_disk_usage(path: str = "", top_n: int = 20) -> str:
    """Analyze disk usage and find largest files/directories"""
    try:
        if not path:
            path = str(USER_HOME)
        
        if not is_safe_path(path):
            return "❌ Error: Access denied - path outside allowed directories"
        
        base_path = Path(path)
        if not base_path.exists():
            return f"❌ Error: Path does not exist: {path}"
        
        results = []
        results.append(f"📊 Disk Usage Analysis: {path}")
        results.append("=" * 50)
        
        # Collect directory sizes
        dir_sizes = []
        file_sizes = []
        
        try:
            for item in base_path.iterdir():
                if item.is_dir():
                    try:
                        size = sum(f.stat().st_size for f in item.rglob('*') if f.is_file())
                        dir_sizes.append((size, item))
                    except (PermissionError, OSError):
                        continue
                elif item.is_file():
                    try:
                        size = item.stat().st_size
                        file_sizes.append((size, item))
                    except (PermissionError, OSError):
                        continue
        except PermissionError:
            return f"❌ Error: Permission denied accessing {path}"
        
        # Sort by size (largest first)
        dir_sizes.sort(reverse=True)
        file_sizes.sort(reverse=True)
        
        # Show largest directories
        results.append(f"\n📂 Largest Directories (Top {min(top_n, len(dir_sizes))}):")
        for i, (size, dir_path) in enumerate(dir_sizes[:top_n]):
            size_mb = size / (1024 * 1024)
            size_gb = size_mb / 1024
            if size_gb > 1:
                results.append(f"  {i+1:2d}. {dir_path.name:<30} {size_gb:8.2f} GB")
            else:
                results.append(f"  {i+1:2d}. {dir_path.name:<30} {size_mb:8.2f} MB")
        
        # Show largest files
        results.append(f"\n📄 Largest Files (Top {min(top_n, len(file_sizes))}):")
        for i, (size, file_path) in enumerate(file_sizes[:top_n]):
            size_mb = size / (1024 * 1024)
            size_gb = size_mb / 1024
            if size_gb > 1:
                results.append(f"  {i+1:2d}. {file_path.name:<30} {size_gb:8.2f} GB")
            else:
                results.append(f"  {i+1:2d}. {file_path.name:<30} {size_mb:8.2f} MB")
        
        # Total size calculation
        total_size = sum(size for size, _ in dir_sizes) + sum(size for size, _ in file_sizes)
        total_gb = total_size / (1024 * 1024 * 1024)
        
        results.append(f"\n📊 Summary:")
        results.append(f"📂 Directories: {len(dir_sizes)}")
        results.append(f"📄 Files: {len(file_sizes)}")
        results.append(f"💾 Total Size: {total_gb:.2f} GB")
        
        return "\n".join(results)
        
    except Exception as e:
        return f"❌ Error: {str(e)}"

    # Check if files exist
    import os
    if not os.path.exists(server_config['command']):
        print(f'⚠️  Warning: Python executable not found: {server_config[\"command\"]}')
    
    if not os.path.exists(server_config['args'][0]):
        print(f'⚠️  Warning: Server script not found: {server_config[\"args\"][0]}')
    
except json.JSONDecodeError as e:
    print(f'❌ Invalid JSON: {e}')
    exit(1)
except Exception as e:
    print(f'❌ Configuration error: {e}')
    exit(1)
"; then
        print_success "Claude Desktop configuration completed successfully"
    else
        print_error "Configuration validation failed"
        return 1
    fi
    
    # Show final configuration summary
    echo
    print_info "📋 Configuration Summary:"
    print_info "  • Config file: $CONFIG_FILE"
    print_info "  • Server name: claude-ops"
    print_info "  • Python: $VENV_PYTHON_PATH"
    print_info "  • Server: $SERVER_SCRIPT_PATH"
    print_info "  • Workspace: $HOME/workspace"
    
    # Create workspace directory
    mkdir -p "$HOME/workspace" 2>/dev/null || true
    
    print_success "Claude Desktop MCP configuration is ready!"
}

# Cleanup on exit
cleanup() {
    if [[ -n "$SERVER_PID" ]] && ps -p $SERVER_PID > /dev/null 2>&1; then
        print_info "Cleaning up background server..."
        kill $SERVER_PID 2>/dev/null || true
    fi
}

trap cleanup EXIT
